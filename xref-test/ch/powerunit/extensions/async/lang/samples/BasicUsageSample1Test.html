22  import java.io.FileNotFoundException;
23  import java.io.FileOutputStream;
24  import java.io.PrintStream;
25  import java.util.Optional;
26  import java.util.concurrent.Callable;
27  import java.util.concurrent.CompletableFuture;
28  import java.util.concurrent.Executors;
29  import java.util.concurrent.TimeUnit;
30  import java.util.function.Consumer;
31  import java.util.function.Predicate;
32  
33  import ch.powerunit.Rule;
34  import ch.powerunit.Test;
35  import ch.powerunit.TestRule;
36  import ch.powerunit.TestSuite;
37  import ch.powerunit.extensions.async.lang.WaitResult;
38  import ch.powerunit.extensions.async.lang.WaitResultBuilder5;
39  
40  public class BasicUsageSample1Test implements TestSuite {
41  
42  	private PrintStream output;
43  
44  	@Rule
45  	public final TestRule rules = testListenerRuleOnStart((c) -> {
46  		try {
47  			output = new PrintStream(
48  					new FileOutputStream("target/" + BasicUsageSample1Test.class.getName() + ".log", true));
49  			output.println(c.getFullTestName() + ":");
50  		} catch (FileNotFoundException e) {
51  			throw new IllegalArgumentException("" + e.getMessage(), e);
52  		}
53  	}).around(after(() -> Optional.ofNullable(output).ifPresent(PrintStream::close)));
54  
55  	private Callable<String> myCallable = () -> "x";
56  
57  	private Callable<String> myCallableWithException = () -> {
58  		throw new Exception("Sample");
59  	};
60  
61  	private <T> Callable<T> displayThread(Callable<T> callable) {
62  		return () -> {
63  			output.println("callable:" + Thread.currentThread().getName());
64  			return callable.call();
65  		};
66  	}
67  
68  	private <T> Predicate<T> displayThread(Predicate<T> predicate) {
69  		return t -> {
70  			output.println("predicate:" + Thread.currentThread().getName());
71  			return predicate.test(t);
72  		};
73  	}
74  
75  	private <T> Consumer<T> displayThreadConsumer(Consumer<T> consumer) {
76  		return t -> {
77  			output.println("consumer:" + Thread.currentThread().getName());
78  			consumer.accept(t);
79  		};
80  	}
81  
82  	@Test
83  	public void testWaitResultOfBasicSample() {
84  		output.println(Thread.currentThread().getName());
85  		//@formatter:off
86  		Optional<String> result = WaitResult.
87  			of(displayThread(myCallable)).
88  			dontIgnoreException().
89  			expecting(displayThread(s->"x".equals(s))).
90  			repeat(2).
91  			every(1000, TimeUnit.MILLISECONDS).
92  			get();
93  		//@formatter:on
94  		assertThat(result).is(optionalIs("x"));
95  	}
96  
97  	@Test
98  	public void testWaitResultOfBasicSampleAsync() {
99  		output.println(Thread.currentThread().getName());
100 		//@formatter:off
101 		CompletableFuture<Optional<String>> future = 
102 			WaitResult.
103 			of(displayThread(myCallable)).
104 			dontIgnoreException().
105 			expecting(displayThread(s->"x".equals(s))).
106 			repeat(2).
107 			every(1000, TimeUnit.MILLISECONDS).
108 			usingDefaultExecutor().
109 			asyncExec();
110 		Optional<String> result = future.join();
111 		//@formatter:on
112 		assertThat(result).is(optionalIs("x"));
113 	}
114 
115 	@Test
116 	public void testWaitResultOfBasicSampleManualAsync() {
117 		output.println(Thread.currentThread().getName());
118 		//@formatter:off
119 		Optional<String> result = WaitResult.
120 			of(displayThread(myCallable)).
121 			dontIgnoreException().
122 			expecting(displayThread(s->"x".equals(s))).
123 			repeat(2).
124 			every(1000, TimeUnit.MILLISECONDS).
125 			using(Executors.newFixedThreadPool(2)).join();
126 		//@formatter:on
127 		assertThat(result).is(optionalIs("x"));
128 	}
129 
130 	@Test
131 	public void testWaitResultOfBasicSampleWithException() {
132 		output.println(Thread.currentThread().getName());
133 		//@formatter:off
134 		WaitResultBuilder5<String> executor = WaitResult.
135 			of(displayThread(myCallableWithException)).
136 			dontIgnoreException().
137 			expecting(displayThread(s->"x".equals(s))).
138 			repeat(2).
139 			every(1000, TimeUnit.MILLISECONDS);
140 		//@formatter:on
141 		assertWhen(executor::get).throwException(instanceOf(AssertionError.class));
142 	}
143 
144 	@Test
145 	public void testWaitResultOfBasicSampleAsyncAccept() {
146 		output.println(Thread.currentThread().getName());
147 		//@formatter:off
148 		WaitResult.
149 			of(displayThread(myCallable)).
150 			dontIgnoreException().
151 			expecting(displayThread(s->"x".equals(s))).
152 			repeat(2).
153 			every(1000, TimeUnit.MILLISECONDS).
154 			usingDefaultExecutor().thenAccept(displayThreadConsumer(x->{})).join();
155 		//@formatter:on
156 	}

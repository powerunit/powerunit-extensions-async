020package ch.powerunit.extensions.async.lang;
021
022/**
023 * This interface can be used to specify retry configuration.
024 * 
025 * @author borettim
026 * @since 1.0.0
027 * @see RetryPolicies
028 */
029public interface RetryPolicy {
030        /**
031         * This is the number of retry to be done.
032         * 
033         * @return the number of retry.
034         */
035        int getCount();
036
037        /**
038         * This is the method called to wait between a retry.
039         * 
040         * @param retry
041         *            the current retry (start at 1). A initial try has been done before
042         *            any retry without wait time.
043         */
044        void sleepBetweenRetry(int retry);
045}

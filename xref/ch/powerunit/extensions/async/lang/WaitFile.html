20  package ch.powerunit.extensions.async.lang;
21  
22  import static ch.powerunit.extensions.async.lang.WaitResult.callableWithToString;
23  import static ch.powerunit.extensions.async.lang.WaitResult.of;
24  import static java.nio.file.StandardWatchEventKinds.ENTRY_CREATE;
25  import static java.nio.file.StandardWatchEventKinds.ENTRY_DELETE;
26  import static java.util.Objects.requireNonNull;
27  import static java.util.stream.Collectors.collectingAndThen;
28  import static java.util.stream.Collectors.toList;
29  
30  import java.nio.file.Path;
31  import java.nio.file.WatchEvent;
32  import java.nio.file.WatchEvent.Kind;
33  import java.util.Collection;
34  import java.util.Collections;
35  import java.util.Objects;
36  import java.util.concurrent.Callable;
37  
38  import ch.powerunit.extensions.async.impl.FilePool;
39  
40  /**
41   * This class provides methods to wait for fileSystem events.
42   * <p>
43   * The goal is to wait for filesystem event, by using the
44   * {@link java.nio.file.WatchService} functionality.
45   * 
46   * @since 1.1.0
47   * @see java.nio.file.WatchService
48   */
49  public final class WaitFile {
50  	private WaitFile() {
51  	}
52  
53  	/**
54  	 * Wait for a folder to have some event.
55  	 * <p>
56  	 * The wait starts at the first try to get the result.
57  	 * <p>
58  	 * For example :
59  	 * 
60  	 * <pre>
61  	 * CompletableFuture&lt;Optional&lt;Collection&lt;WatchEvent&lt;Path&gt;&gt;&gt;&gt; wait = WaitFile
62  	 * 		.eventIn(test, StandardWatchEventKinds.ENTRY_CREATE)
63  	 * 		.expecting(l -&gt; l.stream().map(WatchEvent::context).map(Path::getFileName).map(Path::toString)
64  	 * 				.anyMatch(n -&gt; n.equals("test")))
65  	 * 		.repeat(3).every(Duration.ofMillis(250)).usingDefaultExecutor().asyncExec();
66  	 * </pre>
67  	 * 
68  	 * Defines a 3 tries with a wait time of 250ms, for a creation event, containing
69  	 * at least one event with last part of a path named test.
70  	 * 
71  	 * @param directory
72  	 *            the directory to be verified.
73  	 * @param events
74  	 *            the events to wait for.
75  	 * @return {@link WaitResultBuilder1} the next step of the builder.
76  	 */
77  	@SafeVarargs
78  	public static WaitResultBuilder1<Collection<WatchEvent<Path>>> eventIn(Path directory, Kind<Path>... events) {
79  		requireNonNull(directory, "directory can't be null");
80  		FilePool filePool = new FilePool(directory, events);
81  		return of(filePool, filePool::close);
82  	}
83  
84  	/**
85  	 * Wait for a folder to contains new entry.
86  	 * <p>
87  	 * The wait starts at the first try to get the result.
88  	 * <p>
89  	 * For example :
90  	 * 
91  	 * <pre>
92  	 * CompletableFuture&lt;Optional&lt;Collection&lt;Path&gt;&gt;&gt; wait = WaitFile.newFileIn(test)
93  	 * 		.expecting(l -&gt; l.stream().map(Path::getFileName).map(Path::toString).anyMatch(n -&gt; n.equals("test")))
94  	 * 		.repeat(3).every(Duration.ofMillis(250)).usingDefaultExecutor().asyncExec();
95  	 * </pre>
96  	 * 
97  	 * Defines a 3 tries with a wait time of 250ms, for a list of new file,
98  	 * containing at least one file with last part of a path named test.
99  	 * 
100 	 * 
101 	 * @param directory
102 	 *            the directory to be verified.
103 	 * @return {@link WaitResultBuilder1} the next step of the builder.
104 	 */
105 	public static WaitResultBuilder1<Collection<Path>> newFileIn(Path directory) {
106 		requireNonNull(directory, "directory can't be null");
107 		FilePool filePool = new FilePool(directory, ENTRY_CREATE);
108 		return of(callableWithToString(toPathCollection(filePool), () -> "new file in " + filePool), filePool::close);
109 	}
110 
111 	/**
112 	 * Wait for a folder to contains new entry based on his name.
113 	 * <p>
114 	 * The wait starts at the first try to get the result.
115 	 * <p>
116 	 * For example :
117 	 * 
118 	 * <pre>
119 	 * CompletableFuture&lt;Optional&lt;Path&gt;&gt; wait = WaitFile.newFileNamedIn(test, "test").expectingNotNull().repeat(3)
120 	 * 		.every(Duration.ofMillis(250)).usingDefaultExecutor().asyncExec();
121 	 * </pre>
122 	 * 
123 	 * Defines a 3 tries with a wait time of 25ms, for a file named "test".
124 	 * 
125 	 * @param directory
126 	 *            the directory to be verified.
127 	 * @param name
128 	 *            the expected name
129 	 * @return {@link WaitResultBuilder1} the next step of the builder.
130 	 */
131 	public static WaitResultBuilder1<Path> newFileNamedIn(Path directory, String name) {
132 		requireNonNull(directory, "directory can't be null");
133 		requireNonNull(name, "name can't be null");
134 		FilePool filePool = new FilePool(directory, ENTRY_CREATE);
135 		return of(callableWithToString(toPathByName(toPathCollection(filePool), name),
136 				() -> String.format("New file named %s in %s", name, filePool)), filePool::close);
137 	}
138 
139 	/**
140 	 * Wait for a folder to have entry removed.
141 	 * <p>
142 	 * The wait starts at the first try to get the result.
143 	 * 
144 	 * @param directory
145 	 *            the directory to be verified.
146 	 * @return {@link WaitResultBuilder1} the next step of the builder.
147 	 */
148 	public static WaitResultBuilder1<Collection<Path>> removeFileFrom(Path directory) {
149 		requireNonNull(directory, "directory can't be null");
150 		FilePool filePool = new FilePool(directory, ENTRY_DELETE);
151 		return of(callableWithToString(toPathCollection(filePool), () -> "Removed file from " + directory),
152 				filePool::close);
153 	}
154 
155 	private static Callable<Collection<Path>> toPathCollection(Callable<Collection<WatchEvent<Path>>> callable) {
156 		return () -> callable.call().stream().map(WatchEvent::context)
157 				.collect(collectingAndThen(toList(), Collections::unmodifiableList));
158 	}
159 
160 	private static Callable<Path> toPathByName(Callable<Collection<Path>> callable, String name) {
161 		return () -> callable.call().stream()
162 				.filter(p -> Objects.equals(p.getName(p.getNameCount() - 1).toString(), name)).findFirst().orElse(null);
163 	}
164 
165 }
